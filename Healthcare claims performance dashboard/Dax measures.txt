Approval Rate = 
VAR ApprovedClaims = CALCULATE([Total Claims Processed], 'enhanced_health_insurance_claims'[ClaimStatus] = "Approved")
VAR TotalClaims = [Total Claims Processed]
RETURN DIVIDE(ApprovedClaims, TotalClaims, 0)

Approval TArget = 0.30

Average Claim Amount = AVERAGE('enhanced_health_insurance_claims'[ClaimAmount])

Average Paid Amount per Claim = 
DIVIDE (
    [Total Amount Paid],
    CALCULATE ( [Total Claims Processed], 'enhanced_health_insurance_claims'[ClaimStatus] = "Approved" )
)

Denial Rate = 
VAR DeniedClaims =
    CALCULATE (
        [Total Claims Processed],
        'enhanced_health_insurance_claims'[ClaimStatus] = "Denied"
    )
VAR TotalClaimsForContext = [Total Claims Processed]
RETURN
    DIVIDE ( DeniedClaims, TotalClaimsForContext )
